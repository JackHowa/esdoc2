    private _inspectExportDefaultDeclaration() source

    inspect ExportDefaultDeclaration.

    case1: separated export

    class Foo {}
export default Foo;

    case2: export instance(directly).

    class Foo {}
export default new Foo();

    case3: export instance(indirectly).

    class Foo {}
let foo = new Foo();
export default foo;
